\begin{luacode*}
vibrate_line = function (head, group, size)
  i = 1.5
  for list in node.traverse(head) do
    i = i + 0.13
    if list.id == node.id("hlist")
    then
      list.shift = list.shift - (1200000 * (math.sin (i)))
    end
  end
  return head
end
\end{luacode*}

\directlua{%
  luatexbase.add_to_callback('vpack_filter', vibrate_line, "strict")%
}
