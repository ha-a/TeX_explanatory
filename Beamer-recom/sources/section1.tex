\subsection{Why}
\begin{frame}[fragile]
  \frametitle{\subsecname\ \secname ?}

  \begin{itemize}
    \item \LaTeX コマンドが使える
    \begin{itemize}
      \item 数式がきれい
      \item 全体の構造/階層の理解
    \end{itemize}
    \item スライドの統一性の向上
    \begin{itemize}
      \item 実用的な \verb|theme| が豊富に用意されている
      \item 全体のレイアウト・色・フォントも一括で変更可
      \item OverlayとDynamic効果
    \end{itemize}
    \item 講義ノート$\leftrightarrow$プレゼン資料の変換可
    \item pdf出力$\Rightarrow$表示のデバイスやバージョン依存なし
  \end{itemize}

\end{frame}
